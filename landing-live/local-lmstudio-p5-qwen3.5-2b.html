function setup() { createCanvas(windowWidth, windowHeight); noStroke(); fill(138, 43, 226); // Calming blue color } function draw() { background(250, 250, 250); // Soft off-white background for contrast particleCount = random(width * height); for (i = 0; i < particleCount; i++) { particles[i].pos = translate(random(width), random(height)); particles[i].vel = map(random(1, 2), 0, width, 0, -5); // Gentle horizontal and vertical velocity particles[i].acc = map(random(-1, 1) * 0.05, 0, 1, -0.08, 0); // Subtle acceleration particles[i].life = random(30, 60); } for (i = particleCount - 1; i >= 0; i--) { particles[i].pos.add(particles[i].vel); particles[i].acc += particles[i].acc; if (particles[i].life < 0) { particles.splice(i, 1); } else { particles[i].life--; } } for (i = 0; i < particleCount; i++) { p = particles[i]; background(rgba(p.life * 255, p.life * 255)); // Fade effect based on life if (p.pos.x > width) { p.vel.x *= -1; p.vel.y *= -1; } else if (p.pos.x < 0) { p.vel.x *= -1; p.vel.y *= -1; } p.pos.x += p.vel.x; p.pos.y += p.vel.y; noStroke(); fill(rgba(p.life * 255, p.life * 255)); ellipse(p.pos.x, p.pos.y, 3); } }
