#version 330 core out vec4 FragColor; uniform vec2 iResolution; uniform float iTime; void mainImage( out vec4 fragColor, in vec2 fragCoord ) { vec2 uv = fragCoord.xy / iResolution.xy; float t = iTime * 0.75; // Create plasma pattern float v = 0.0; v += sin((uv.x * 10.0) + t); v += sin((uv.y * 10.0) + t / 2.0); v += sin((uv.x * 10.0 + uv.y * 10.0) + t / 2.0); float cx = uv.x + 0.5 * sin(t / 3.0); float cy = uv.y + 0.5 * cos(t / 2.0); v += sin(sqrt(100.0 * (cx * cx + cy * cy)) + t); // Normalize the value roughly v = v / 4.0; // Generate Color based on value vec3 col = 0.5 + 0.5 * cos(6.28318 * (v + vec3(0.0, 0.333, 0.667))); fragColor = vec4(col, 1.0); }
