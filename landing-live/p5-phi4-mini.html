let particles = [];

class Particle {
  constructor(x, y) {
    this.pos = createVector(x, y);
    // Initialize velocity with slight randomness for initial spread
    this.vel = p5.Vector.random2D().mult(random(0.1, 1));
    this.acc = createVector(0, 0);
    this.lifespan = 255;
    this.size = random(3, 6);
    this.color = color(100, 150, 255, 150); // Calming blue
  }

  applyForce(force) {
    this.acc.add(force);
  }

  update() {
    // Apply natural flow using sine wave based on position for smooth movement
    let flowX = sin(this.pos.y * 0.01 + frameCount * 0.005) * 0.3;
    let flowY = cos(this.pos.x * 0.01 + frameCount * 0.005) * 0.3;

    // Add flow force to acceleration
    this.applyForce(createVector(flowX, flowY));

    this.vel.add(this.acc);
    this.vel.limit(2); // Keep velocity manageable
    this.pos.add(this.vel);
    this.acc.mult(0); // Reset acceleration

    this.lifespan -= 1.5;
  }

  show() {
    stroke(100, 150, 255, this.lifespan);
    strokeWeight(this.size);
    point(this.pos.x, this.pos.y);
  }

  isDone() {
    return this.lifespan < 0;
  }
}

function setup() {
  createCanvas(800, 500);
  colorMode(HSB, 360, 100, 100, 100);
  background(220);
}

function draw() {
  // Create a fading background effect to simulate trails (calming)
  background(220, 100, 100, 10); 

  // 1. Update and Draw Particles
  for (let i = particles.length - 1; i >= 0; i--) {
    let p = particles[i];
    p.update();
    p.show();

    if (p.isDone()) {
      particles.splice(i, 1);
    }
  }

  // 2. Spawn new particles gently
  if (frameCount % 5 === 0 && particles.length < 300) {
    let x = random(width);
    let y = random(height);
    particles.push(new Particle(x, y));
  }
}
