let particles = []; const numParticles = 150; const blueColorHue = 200; // Blue range in HSB class Particle { constructor(x, y) { this.pos = createVector(x, y); // Initialize velocity with a slight starting push this.vel = createVector(random(-1, 1), random(-1, 1)); this.acc = createVector(0, 0); this.lifespan = 255; this.size = random(3, 6); } applyForce(force) { this.acc.add(force); } update(time) { // 1. Flowing/Guiding Force (Using Sine Waves based on position and time) let noiseFactorX = map(this.pos.x, 0, width, 0, 1); let noiseFactorY = map(this.pos.y, 0, height, 0, 1); // Create a gentle, undulating force field let flowForceX = cos(noiseFactorX * 10 + time * 0.005) * 0.15; let flowForceY = sin(noiseFactorY * 10 + time * 0.005) * 0.15; let flowForce = createVector(flowForceX, flowForceY); this.applyForce(flowForce); // 2. Standard Physics Update this.vel.add(this.acc); this.vel.limit(1.5); // Limit speed for a gentle feel this.pos.add(this.vel); this.acc.mult(0); // Reset acceleration // 3. Lifespan and Friction (Slow decay) this.lifespan -= 1.5; this.vel.mult(0.98); // Slight drag/friction } display() { // Use HSB for easy color control colorMode(HSB, 360, 100, 100, 100); // Calming Blue: Hue around 200, moderate saturation/brightness let blue = color(blueColorHue, 70, 70, this.lifespan / 255); // Draw the particle, size fades with lifespan let currentSize = map(this.lifespan, 0, 255, 0, this.size * 1.5); stroke(blue); strokeWeight(currentSize); noFill(); ellipse(this.pos.x, this.pos.y, currentSize, currentSize); } } function setup() { createCanvas(windowWidth, windowHeight); background(10, 10, 15); // Very dark, deep background // Initialize particles spread across the screen for (let i = 0; i < numParticles; i++) { let x = random(width); let y = random(height); particles.push(new Particle(x, y)); } } function draw() { // Create a semi-transparent overlay to create a trailing, flowing effect background(10, 10, 15, 30); let time = frameCount * 0.01; for (let i = 0; i < particles.length; i++) { let p = particles[i]; p.update(time); p.display(); } // Handle particle respawn/reset if they fade completely for (let i = particles.length - 1; i >= 0; i--) { if (particles[i].lifespan < 0) { // Respawn them gently near the center or edge particles[i] = new Particle(random(width), random(height)); } } } function windowResized() { resizeCanvas(windowWidth, windowHeight); }
