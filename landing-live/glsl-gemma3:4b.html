#version 330

uniform vec2 u_resolution;
uniform float u_time;

void main() {
    // Normalized UV coordinates
    vec2 uv = gl_FragCoord.xy / u_resolution.xy;
    vec2 p = uv * 2.0 - 1.0; // Map to [-1, 1] range
    float t = u_time * 0.5;

    // --- Plasma Wave Generation ---

    // 1. Base wave pattern using sine functions modulated by time and space
    float wave1 = sin(p.x * 5.0 + t * 1.5) * 0.5 + 0.5;
    float wave2 = cos(p.y * 4.0 + t * 0.8) * 0.5 + 0.5;
    float wave3 = sin((p.x + p.y) * 3.0 + t * 2.0) * 0.3 + 0.7;

    // Combine waves to create a dynamic intensity map
    float plasma_intensity = (wave1 + wave2 + wave3) / 3.0;

    // 2. Adding a refractive/swirling element using distance and time
    float dist_sq = dot(p, p);
    float swirl = sin(sqrt(dist_sq) * 10.0 + t * 1.0) * 0.2 + 0.8;

    // Final composite factor, blending the structured waves with the swirl
    float final_factor = plasma_intensity * 0.6 + swirl * 0.4;

    // --- Color Mapping (Simulating plasma glow/shifting colors) ---

    // Map the factor (0.0 to 1.0) across multiple color channels over time
    float r = sin(final_factor * 10.0 + t * 0.5) * 0.5 + 0.5;
    float g = cos(final_factor * 10.0 + t * 0.8) * 0.5 + 0.5;
    float b = sin(final_factor * 10.0 + t * 1.2) * 0.5 + 0.5;

    // Apply overall brightness modulation based on the original factor
    float glow = pow(final_factor, 1.5) * 1.5;

    // Output color, glowing and animated
    gl_FragColor = vec4(r * glow, g * glow, b * glow, 1.0);
}
