let particles = []; const NUM_PARTICLES = 200; function setup() { createCanvas(windowWidth, windowHeight); colorMode(HSB, 360, 100, 100, 100); for (let i = 0; i < NUM_PARTICLES; i++) { particles.push(new Particle()); } } function draw() { background(220, 30, 10, 20); // subtle blue‑tinted trail for (let p of particles) { p.update(); p.display(); } } class Particle { constructor() { this.pos = createVector(random(width), random(height)); this.vel = createVector(random(-0.2, 0.2), random(-0.2, 0.2)); this.size = random(2, 6); this.hue = random(190, 230); // calming blue range this.sat = random(50, 80); this.bri = random(70, 100); this.alpha = random(40, 80); } update() { this.vel.x += random(-0.01, 0.01); this.vel.y += random(-0.01, 0.01); this.vel.mult(0.99); this.pos.add(this.vel); // wrap around edges this.pos.x = (this.pos.x + width) % width; this.pos.y = (this.pos.y + height) % height; } display() { noStroke(); fill(this.hue, this.sat, this.bri, this.alpha); ellipse(this.pos.x, this.pos.y, this.size); } } function windowResized() { resizeCanvas(windowWidth, windowHeight); }
