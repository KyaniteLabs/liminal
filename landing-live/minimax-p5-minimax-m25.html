function setup() { createCanvas(windowWidth, windowHeight); noStroke(); for (let i = 0; i < 150; i++) { particles.push(new Particle()); } } function draw() { background(10, 20, 40, 50); for (let p of particles) { p.update(); p.display(); } } function windowResized() { resizeCanvas(windowWidth, windowHeight); } class Particle { constructor() { this.pos = createVector(random(width), random(height)); this.vel = p5.Vector.random2D().mult(random(0.2, 0.5)); this.acc = createVector(0, 0); this.size = random(3, 8); this.alpha = random(100, 200); this.color = color(random(50, 150), random(100, 200), 255, this.alpha); } update() { let mouse = createVector(mouseX, mouseY); let dir = p5.Vector.sub(mouse, this.pos); let d = dir.mag(); if (d < 200) { dir.setMag(0.02); this.acc.add(dir); } this.vel.add(this.acc); this.vel.limit(1.5); this.pos.add(this.vel); this.acc.mult(0); if (this.pos.x < 0) this.pos.x = width; if (this.pos.x > width) this.pos.x = 0; if (this.pos.y < 0) this.pos.y = height; if (this.pos.y > height) this.pos.y = 0; } display() { fill(this.color); ellipse(this.pos.x, this.pos.y, this.size); } } let particles = [];
