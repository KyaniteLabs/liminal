let particles = []; const NUM_PARTICLES = 150; const BLUE_HUE = 200; // A soothing blue hue class Particle { constructor(x, y) { this.pos = createVector(x, y); this.vel = p5.Vector.random2D().mult(random(0.1, 0.5)); this.acc = createVector(0, 0); this.lifespan = 255; this.radius = random(1, 3); } applyForce(force) { this.acc.add(force); } update() { // Apply gentle gravity/drift (vertical calming pull) let drift = createVector(0, 0.01); this.applyForce(drift); this.vel.add(this.acc); this.vel.limit(1.5); // Keep movement gentle this.pos.add(this.vel); this.acc.mult(0); // Reset acceleration this.lifespan -= 2; // Slow fade } show() { let alpha = this.lifespan; let color = color(BLUE_HUE, 80, 100, alpha); stroke(color); strokeWeight(this.radius); point(this.pos.x, this.pos.y); } } function setup() { createCanvas(windowWidth, windowHeight); background(0); // Initialize particles across the screen for (let i = 0; i < NUM_PARTICLES; i++) { let x = random(width); let y = random(height); particles.push(new Particle(x, y)); } } function draw() { // Create a faint, semi-transparent background draw to simulate lingering trails background(0, 0, 0, 15); for (let i = particles.length - 1; i >= 0; i--) { let p = particles[i]; // 1. Update position p.update(); // 2. Show particle p.show(); // 3. Remove dead particles and respawn new ones if (p.lifespan < 0) { particles.splice(i, 1); // Respawn particle at a random, calming location particles.push(new Particle(random(width), random(height))); } } } function windowResized() { resizeCanvas(windowWidth, windowHeight); }
