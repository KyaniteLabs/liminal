import * as THREE from 'three'; // Setup scene const scene = new THREE.Scene(); scene.background = new THREE.Color(0x111122); // Camera setup const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 1000); camera.position.set(2, 2, 5); // Renderer setup const renderer = new THREE.WebGLRenderer({ antialias: true }); renderer.setSize(window.innerWidth, window.innerHeight); document.body.appendChild(renderer.domElement); // --- Lighting --- // Ambient Light: Soft overall illumination const ambientLight = new THREE.AmbientLight(0x404040, 2); // Soft white light scene.add(ambientLight); // Directional Light: Simulates sun/key light, creating strong shadows const directionalLight = new THREE.DirectionalLight(0xffffff, 2); directionalLight.position.set(1, 5, 10); scene.add(directionalLight); // Point Light: Adds localized, warm glow (e.g., a lamp) const pointLight = new THREE.PointLight(0xffa500, 1, 50); // Orange/Gold color, intensity 1, range 50 pointLight.position.set(-5, 2, -5); scene.add(pointLight); // Hemisphere Light: Simulates skylight, good for soft ground bounce const hemiLight = new THREE.HemisphereLight(0xffffff, 0x444466, 1); // Sky color to ground color scene.add(hemiLight); // --- Geometry (The Cube) --- // Create the BoxGeometry const geometry = new THREE.BoxGeometry(1, 1, 1); // Create the MeshPhongMaterial for better light interaction (specular highlights) const material = new THREE.MeshPhongMaterial({ color: 0xcccccc, // Light grey base color specular: 0x333333, // Specular color (how shiny it is) shininess: 80 // How sharp the highlight is }); // Create the mesh const cube = new THREE.Mesh(geometry, material); scene.add(cube); // --- Animation Loop --- function animate() { requestAnimationFrame(animate); // Rotation logic cube.rotation.x += 0.005; cube.rotation.y += 0.01; renderer.render(scene, camera); } // Handle window resizing window.addEventListener('resize', () => { camera.aspect = window.innerWidth / window.innerHeight; camera.updateProjectionMatrix(); renderer.setSize(window.innerWidth, window.innerHeight); }); animate();
