const NUM_PARTICLES = 100;
const BLUE_HUE = 200; // A calming blue value

class Particle {
  constructor(x, y) {
    this.pos = createVector(x, y);
    // Initial velocity with slight randomness
    this.vel = createVector(random(-1, 1), random(-1, 1));
    this.acc = createVector(0, 0);
    this.lifespan = random(150, 255); // Controls fade rate
    this.radius = random(1, 3);
  }

  applyForce(force) {
    this.acc.add(force);
  }

  update() {
    // 1. Apply Flow Force (Sine wave based flow)
    // This creates a gentle horizontal push based on time and Y position
    let flowX = sin(this.pos.y * 0.01 + frameCount * 0.01) * 0.3;
    let flowY = cos(this.pos.x * 0.01 + frameCount * 0.01) * 0.1;
    
    let flowForce = createVector(flowX, flowY);
    this.applyForce(flowForce);

    // 2. Update Velocity and Position
    this.vel.add(this.acc);
    this.vel.mult(0.98); // Damping/Air resistance
    this.pos.add(this.vel);
    this.acc.mult(0); // Reset acceleration

    // 3. Fade lifespan (for gentle disappearance)
    this.lifespan -= 1.5;
  }

  display() {
    // Set color to a calming blue that fades
    stroke(BLUE_HUE, 150, 255, this.lifespan);
    strokeWeight(this.radius * (this.lifespan / 255));
    
    // Draw the particle
    point(this.pos.x, this.pos.y);
  }

  isDead() {
    return this.lifespan < 0;
  }
}
