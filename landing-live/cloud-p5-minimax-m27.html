let particles = []; const numParticles = 250; function setup() { createCanvas(windowWidth, windowHeight); colorMode(HSB, 360, 100, 100, 255); for (let i = 0; i < numParticles; i++) { particles.push(new Particle()); } } function draw() { // soft trail effect background(220, 60, 20, 25); for (let p of particles) { p.update(); p.display(); } } class Particle { constructor() { this.pos = createVector(random(width), random(height)); this.vel = createVector(random(-0.3, 0.3), random(-0.6, -0.1)); this.size = random(2, 9); this.hue = random(190, 240); this.sat = random(60, 90); this.bri = random(80, 100); this.alpha = random(80, 180); } update() { // gentle noise-based drift let angle = noise(this.pos.x * 0.001, this.pos.y * 0.001, frameCount * 0.005) * TWO_PI * 2; this.vel.x += cos(angle) * 0.015; this.vel.y += sin(angle) * 0.015; this.vel.limit(0.6); this.pos.add(this.vel); // wrap around edges if (this.pos.x < 0) this.pos.x = width; if (this.pos.x > width) this.pos.x = 0; if (this.pos.y < 0) this.pos.y = height; if (this.pos.y > height) this.pos.y = 0; } display() { noStroke(); fill(this.hue, this.sat, this.bri, this.alpha); ellipse(this.pos.x, this.pos.y, this.size); } } function windowResized() { resizeCanvas(windowWidth, windowHeight); }
