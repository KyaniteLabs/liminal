let particles = []; const numParticles = 300; const noiseScale = 0.005; function setup() { createCanvas(windowWidth, windowHeight); colorMode(HSB, 360, 100, 100, 100); for (let i = 0; i < numParticles; i++) { particles.push(new Particle()); } } function draw() { // Subtle fade for soft trails blendMode(BLEND); background(220, 30, 10, 20); for (let p of particles) { p.update(); p.display(); } } class Particle { constructor() { this.pos = createVector(random(width), random(height)); this.vel = createVector(0, 0); this.prevPos = this.pos.copy(); // Vary hue around calm blues (200‑240) this.hue = random(200, 240); this.saturation = random(40, 70); this.brightness = random(70, 95); this.maxSpeed = random(0.5, 1.5); this.noiseOffset = random(1000); } update() { // Use Perlin noise to create a flowing field let angle = noise( this.pos.x * noiseScale + this.noiseOffset, this.pos.y * noiseScale + this.noiseOffset ) * TWO_PI * 2; let force = p5.Vector.fromAngle(angle); this.vel.add(force); this.vel.limit(this.maxSpeed); this.prevPos = this.pos.copy(); this.pos.add(this.vel); } display() { stroke(this.hue, this.saturation, this.brightness, 80); strokeWeight(2); line(this.prevPos.x, this.prevPos.y, this.pos.x, this.pos.y); } } function windowResized() { resizeCanvas(windowWidth, windowHeight); }
