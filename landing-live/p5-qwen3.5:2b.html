let particles = [];
const NUM_PARTICLES = 150;

class Particle {
  constructor(x, y) {
    this.pos = createVector(x, y);
    this.vel = createVector(0, 0);
    this.acc = createVector(0, 0);
    this.lifespan = 255;
    this.radius = random(1, 3);
  }

  applyForce(force) {
    this.acc.add(force);
  }

  update() {
    this.vel.add(this.acc);
    this.vel.mult(0.98); // Slight drag/friction
    this.pos.add(this.vel);
    this.acc.mult(0); // Reset acceleration

    // Fade out slowly
    this.lifespan -= 1.5;
  }

  show() {
    stroke(100, 180, 255, this.lifespan); // Calming blue color
    strokeWeight(this.radius);
    point(this.pos.x, this.pos.y);
  }
}

function setup() {
  createCanvas(windowWidth, windowHeight);
  background(10, 10, 30); // Very dark, calming background
  
  // Initialize particles spread across the screen
  for (let i = 0; i < NUM_PARTICLES; i++) {
    let x = random(width);
    let y = random(height);
    particles.push(new Particle(x, y));
  }
}

function draw() {
  // Create a faint, trailing background effect instead of clearing completely
  background(10, 10, 30, 50); 

  let flowSpeed = 0.005;
  let flowScale = 0.01;

  for (let i = 0; i < particles.length; i++) {
    let p = particles[i];

    // 1. Calculate flow force using Perlin Noise
    // Noise depends on position and time (frameCount) to create movement
    let noiseValX = noise(p.pos.x * flowScale, p.pos.y * flowScale, frameCount * flowSpeed);
    let noiseValY = noise(p.pos.x * flowScale * 0.8 + 100, p.pos.y * flowScale, frameCount * flowSpeed);

    // Map noise values (0 to 1) to directional movement components
    let angle = map(noiseValX, 0, 1, 0, TWO_PI);
    let force = p5.Vector.fromAngle(angle).mult(0.5);
    
    // Add a slight vertical influence based on the second noise check
    force.add(createVector(0, map(noiseValY, 0, 1, -0.5, 0.5)));

    p.applyForce(force);

    // 2. Update and Draw
    p.update();
    p.show();
    
    // 3. Respawn dead particles (optional, keeps the system dense)
    if (p.lifespan < 0) {
      particles[i] = new Particle(random(width), random(height));
    }
  }
}

function windowResized() {
  resizeCanvas(windowWidth, windowHeight);
}
