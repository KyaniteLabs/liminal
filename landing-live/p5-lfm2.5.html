let particles = [];
const NUM_PARTICLES = 100;
const attractor = { x: 0, y: 0 };

// --- Particle Class ---
class Particle {
    constructor(x, y) {
        this.pos = createVector(x, y);
        this.vel = createVector(random(-0.5, 0.5), random(-0.5, 0.5));
        this.acc = createVector(0, 0);
        this.r = random(1, 3);
        this.hue = random(180, 220); // Blue range (Hue in HSB)
        this.life = 255;
        this.maxLife = 255;
    }

    applyForce(force) {
        this.acc.add(force);
    }

    update() {
        // Apply acceleration to velocity
        this.vel.add(this.acc);
        this.vel.limit(1.5); // Limit speed for a gentle flow
        
        // Apply velocity to position
        this.pos.add(this.vel);
        
        // Reset acceleration for the next frame
        this.acc.mult(0);
        
        // Decay life
        this.life -= 1.5;
    }

    show() {
        // Use HSB for controlled blue coloring
        colorMode(HSB, 360, 100, 100, 100);
        let alpha = map(this.life, 0, this.maxLife, 0, 100);
        stroke(this.hue, 80, 90, alpha);
        strokeWeight(this.r);
        point(this.pos.x, this.pos.y);
    }

    isDead() {
        return this.life < 0;
    }
}

// --- Setup ---
function setup() {
    createCanvas(windowWidth, windowHeight);
    background(0, 0, 0, 0); // Transparent background
    colorMode(HSB, 360, 100, 100, 100);

    // Initialize particles across the screen
    for (let i = 0; i < NUM_PARTICLES; i++) {
        particles.push(new Particle(random(width), random(height)));
    }
}

// --- Draw Loop ---
function draw() {
    // Semi-transparent overlay for the fading "trailing" effect
    background(0, 0, 0, 5); 

    // Calculate flow force using Perlin Noise
    let forceX = noise(frameCount * 0.01, map(sin(frameCount * 0.05), -1, 1, 0, width));
    let forceY = noise(frameCount * 0.01, map(cos(frameCount * 0.05), -1, 1, 0, height));
    
    // Normalize noise output (0 to 1) to a vector force
    let flowForce = createVector(
        map(forceX, 0, 1, -0.5, 0.5), 
        map(forceY, 0, 1, -0.5, 0.5)
    );
    
    for (let i = particles.length - 1; i >= 0; i--) {
        let p = particles[i];

        // 1. Apply Flow Force (Perlin Noise)
        p.applyForce(p.createVector(
            sin(p.pos.x * 0.001 + frameCount * 0.01) * 0.2,
            cos(p.pos.y * 0.001 + frameCount * 0.01) * 0.2
        ));
        
        // 2. Apply global flow force (Perlin noise)
        p.applyForce(p.createVector(forceX * 0.1, forceY * 0.1));

        p.update();
        p.show();

        // Respawn dead particles
        if (p.isDead()) {
            particles.splice(i, 1);
            particles.push(new Particle(random(width), random(height)));
        }
    }
}

// --- Window Resizing ---
function windowResized() {
    resizeCanvas(windowWidth, windowHeight);
    // Re-initialize particles to fit new dimensions
    particles = [];
    for (let i = 0; i < NUM_PARTICLES; i++) {
        particles.push(new Particle(random(width), random(height)));
    }
}
