let particles = []; let numParticles = 200; function setup() { createCanvas(windowWidth, windowHeight); noStroke(); for (let i = 0; i < numParticles; i++) { particles.push(new Particle()); } } function draw() { background(10, 20, 40, 50); for (let p of particles) { p.update(); p.display(); } } function windowResized() { resizeCanvas(windowWidth, windowHeight); } class Particle { constructor() { this.reset(); this.pos = createVector(random(width), random(height)); } reset() { this.pos = createVector(random(width), height + 10); this.vel = createVector(random(-0.5, 0.5), random(-1, -3)); this.acc = createVector(0, 0); this.size = random(2, 8); this.alpha = random(50, 200); this.noiseOffset = random(1000); } update() { let angle = noise(this.noiseOffset) * TWO_PI * 2; let flowForce = p5.Vector.fromAngle(angle); flowForce.mult(0.05); this.acc.add(flowForce); this.vel.add(this.acc); this.vel.limit(2); this.pos.add(this.vel); this.acc.mult(0); this.noiseOffset += 0.01; if (this.pos.y < -10 || this.pos.x < -10 || this.pos.x > width + 10) { this.reset(); } } display() { let blue = color(100, 150, 255, this.alpha); fill(blue); ellipse(this.pos.x, this.pos.y, this.size); } }
