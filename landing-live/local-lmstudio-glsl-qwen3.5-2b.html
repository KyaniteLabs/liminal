void main() { // Set up the fragment output to use a fixed precision color for smooth animation gl_FragColor = vec4(0.0, 0.5275, 1.0, 0.9); // Define the main shader function that will be called by the GPU void plasmaShader() { // Get the fragment position in normalized coordinates (x: -1 to 1) vec2 uv = gl_FragCoord.xy / u_resolution.xy; // Create a base color and adjust it based on position for an "abstract" look vec3 baseColor = vec3(0.0, 0.5456, 1.0); // Add a slight gradient offset to make the colors shift across the screen uv += vec2(0.0709, -0.0833) * sin(gl_LocTime * 1.0); // Calculate a colorful, abstract mix of base color and UV coordinates vec3 finalColor = mix(baseColor, uv, pow(abs(uv.x), 2.0)); // Mix the result with white to create an inner glow effect (subtle) gl_FragColor = mix(vec1(1.0, 1.0, 1.0), vec3(finalColor * 0.95 + baseColor * 0.05), 0.2); // Send the color to the GPU for display (note: gl_FragColor is not set here) } // Set the fragment color based on the shader function result gl_FragColor = vec4(gl_FragColor.xyz, 1.0); // Execute the shader function plasmaShader(); }
