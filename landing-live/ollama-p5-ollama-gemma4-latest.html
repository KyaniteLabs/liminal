let particles = []; const NUM_PARTICLES = 100; class Particle { constructor(x, y) { this.pos = createVector(x, y); // Give it a gentle initial velocity this.vel = p5.Vector.random2D().mult(random(0.1, 0.5)); this.acc = createVector(0, 0); this.lifespan = 255; this.radius = random(1, 3); } applyForce(force) { this.acc.add(force); } update() { // Apply drag/friction to slow movement naturally this.vel.mult(0.98); // Update position and velocity this.vel.add(this.acc); this.pos.add(this.vel); this.acc.mult(0); // Reset acceleration // Calculate fading lifespan this.lifespan -= 1.5; } show() { // Calming blue color scheme (e.g., cyan/blue) stroke(100, 150, 255, this.lifespan); strokeWeight(this.radius); point(this.pos.x, this.pos.y); } isDead() { return this.lifespan < 0; } } function setup() { createCanvas(windowWidth, windowHeight); background(10, 10, 50); // Deep, dark blue background // Initialize particles spread across the screen for (let i = 0; i < NUM_PARTICLES; i++) { let x = random(width); let y = random(height); particles.push(new Particle(x, y)); } } function draw() { // Create a slight trail effect by drawing a semi-transparent rectangle over the background background(10, 10, 50, 30); // --- Flowing Force Simulation --- // Use sine/cosine waves to create a guiding, flowing force field let timeFactor = frameCount * 0.01; let force = p5.Vector.create( cos(timeFactor + 0), sin(timeFactor + 1) ); force.normalize(); force.mult(0.01); for (let i = particles.length - 1; i >= 0; i--) { let p = particles[i]; // Apply the general flowing force p.applyForce(force); // Add slight attraction/repulsion based on position to keep them somewhat organized let centerForce = p5.Vector.sub(createVector(width / 2, height / 2), p.pos); centerForce.setMag(0.005); p.applyForce(centerForce); p.update(); p.show(); if (p.isDead()) { // Re-spawn dead particles at a random location for continuous flow particles.splice(i, 1); particles.push(new Particle(random(width), random(height))); } } } function windowResized() { resizeCanvas(windowWidth, windowHeight); background(10, 10, 50); }
