precision highp float; uniform vec2 u_resolution; uniform float u_time; out vec4 fragColor; void main() { // Normalized coordinates from [0, 1] vec2 uv = gl_FragCoord.xy / u_resolution.xy; // --- Time and Space Modulation --- // Scale time and coordinates to create periodic patterns vec2 p = uv * 8.0 + vec2(u_time * 0.1, u_time * 0.08); float t = u_time * 0.5; // --- Plasma Wave Function (Using Sinusoidal combinations) --- // Create multiple interacting waves for complexity float wave1 = sin(p.x * 3.0 + t * 1.5) * 0.5 + 0.5; float wave2 = cos(p.y * 2.5 - t * 0.8) * 0.5 + 0.5; float wave3 = sin(p.x * 1.2 + p.y * 0.8 + t * 2.0) * 0.3 + 0.7; // Combine waves to modulate intensity/color factors float plasma_factor = (wave1 + wave2 + wave3) / 2.0; // --- Color Generation (Vibrant, electric colors) --- // Use the factor to drive color shifts vec3 color_a = vec3(0.1, 1.0, 1.0); // Cyan/Blue vec3 color_b = vec3(1.0, 0.7, 0.1); // Yellow/Orange vec3 color_c = vec3(1.0, 0.1, 1.0); // Magenta // Blend colors based on the modulation factor vec3 final_color = mix(color_a, color_b, plasma_factor * 0.8 + 0.2); final_color = mix(final_color, color_c, sin(p.x * 0.5 + t * 0.5) * 0.3 + 0.7); // Add a slight glow/intensity modulation float intensity = pow(plasma_factor, 2.0) * 1.5 + 0.5; final_color *= intensity; // Output the final color fragColor = vec4(final_color, 1.0); }
