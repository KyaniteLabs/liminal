let particles = []; const NUM_PARTICLES = 150; const BLUE_BASE_HUE = 200; // Blue in HSB color mode class Particle { constructor(x, y) { this.pos = createVector(x, y); // Gentle, slow initial velocity this.vel = p5.Vector.random2D().mult(random(0.1, 0.5)); this.acc = createVector(0, 0); this.lifespan = 255; this.radius = random(1, 3); } applyForce(force) { this.acc.add(force); } update() { // Apply gentle drag/damping to maintain a slow, floating feel this.vel.mult(0.98); this.acc.mult(0); // Reset acceleration // Apply natural, subtle movement force (simulating gentle current) let currentX = this.pos.x; let currentY = this.pos.y; // Gentle horizontal drift based on time and position let driftX = sin(frameCount * 0.005 + currentY * 0.01) * 0.1; // Gentle vertical drift based on time let driftY = cos(frameCount * 0.004 + currentX * 0.01) * 0.05; this.applyForce(createVector(driftX, driftY)); this.vel.add(this.acc); this.pos.add(this.vel); // Fade and shrink over time this.lifespan -= 1.5; this.radius = map(this.lifespan, 255, 0, 3, 0.5); } display() { // Use the particle's lifespan for transparency let alpha = map(this.lifespan, 255, 0, 20, 100); fill(BLUE_BASE_HUE, 80, 90, alpha); noStroke(); ellipse(this.pos.x, this.pos.y, this.radius, this.radius); } isDead() { return this.lifespan < 0; } } function setup() { createCanvas(windowWidth, windowHeight); colorMode(HSB, 360, 100, 100, 100); background(0, 0, 0, 0); // Start with fully transparent background // Initialize particles spread across the screen for (let i = 0; i < NUM_PARTICLES; i++) { let x = random(width); let y = random(height); particles.push(new Particle(x, y)); } } function draw() { // Create a semi-transparent background overlay instead of a full clear. // This creates the ethereal trailing effect essential for "calming." background(0, 0, 0, 10); for (let i = particles.length - 1; i >= 0; i--) { let p = particles[i]; p.update(); p.display(); if (p.isDead()) { // Reset dead particles to a new location particles.splice(i, 1); let newP = new Particle(random(width), random(height)); particles.push(newP); } } } function windowResized() { resizeCanvas(windowWidth, windowHeight); }
