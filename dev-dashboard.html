Guardrail Initialization

import { initializeGuardrailSystem } from './guardrails/index.js';

const system = initializeGuardrailSystem({
  shadowMode: false,              // false = enforce, true = log only
  defaultTier: GuardrailTier.ENFORCING,
  telemetry: true,
});

// Register additional guardrails
system.registry.register(new TypeCheckGuardrail({
  maxErrors: 10,
  autoFix: false,
  failOnError: true,
}));

// Evaluate
const result = await system.registry.evaluate(context);
