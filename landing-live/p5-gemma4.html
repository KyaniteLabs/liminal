let particles = [];
const NUM_PARTICLES = 150;
let noiseScale = 0.005;

class Particle {
  constructor(x, y) {
    this.pos = createVector(x, y);
    this.vel = createVector(0, 0);
    this.acc = createVector(0, 0);
    this.radius = random(1, 3);
    this.lifespan = 255;
    this.baseColor = color(100, 150, 255, 150); // Calming blue base
  }

  applyForce(force) {
    this.acc.add(force);
  }

  update() {
    // 1. Apply flow force derived from Perlin Noise
    let noiseValX = noise(this.pos.x * noiseScale, this.pos.y * noiseScale, frameCount * 0.001);
    let noiseValY = noise(this.pos.x * noiseScale, this.pos.y * noiseScale, frameCount * 0.001 + 100);

    // Calculate directional forces based on noise (smoother motion)
    let flowForceX = map(noiseValX, 0, 1, -0.5, 0.5);
    let flowForceY = map(noiseValY, 0, 1, -0.5, 0.5);

    let flow = createVector(flowForceX, flowForceY);
    
    // Dampening the movement slightly to maintain a 'floating' feel
    this.vel.add(flow);
    this.vel.mult(0.95); // Friction/Damping
    
    this.acc.add(flow); // Add flow to acceleration for this frame
    
    this.vel.add(this.acc);
    this.pos.add(this.vel);
    this.acc.mult(0); // Reset acceleration

    // Fade out over time
    this.lifespan -= 1.5;
  }

  display() {
    stroke(100, 150, 255, this.lifespan);
    strokeWeight(this.radius);
    noFill();
    ellipse(this.pos.x, this.pos.y, this.radius * 2, this.radius * 2);
  }

  isDead() {
    return this.lifespan < 0;
  }
}

function setup() {
  createCanvas(windowWidth, windowHeight);
  angleMode(DEGREES);
  
  // Initialize particles spread across the screen
  for (let i = 0; i < NUM_PARTICLES; i++) {
    let x = random(width);
    let y = random(height);
    particles.push(new Particle(x, y));
  }
}

function draw() {
  // Calming background: Very light, desaturated blue/cyan
  background(230, 245, 255, 50); 

  for (let i = particles.length - 1; i >= 0; i--) {
    let p = particles[i];
    
    p.update();
    p.display();

    if (p.isDead()) {
      // Respawn particle at a random location if it fades away
      particles.splice(i, 1);
      particles.push(new Particle(random(width), random(height)));
    }
  }
}

function windowResized() {
  resizeCanvas(windowWidth, windowHeight);
}
